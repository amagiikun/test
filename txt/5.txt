#include <stdio.h>

int isPrime(int number) {
    int i;

    if (number <= 1) {
        return 0; // 不是素数
    }

    if (number == 2) {
        return 1; // 是素数
    }

    if (number % 2 == 0) {
        return 0; // 不是素数
    }

    for (i = 3; i * i <= number; i += 2) {
        if (number % i == 0) {
            return 0; // 不是素数
        }
    }

    return 1; // 是素数
}

int main() {
    int number;

    printf("请输入一个整数：");
    scanf("%d", &number);

    if (isPrime(number)) {
        printf("%d 是素数\n", number);
    } else {
        printf("%d 不是素数\n", number);
    }

    return 0;
}
